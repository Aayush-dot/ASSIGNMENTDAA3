SELECTION SORT


#include <iostream>
using namespace std;
void selectionSort(int A[],int n)
{
for(int i=0;i < n-1;i++)
{
int min=A[i];
int minindx=i;
for(int j=i+1;j if(min > A[j])
{
min=A[j];
minindx=j;
} } swap(A[i],A[minindx]);
} }
int main()
{
int n;
cout<<"Enter number of elements\n";
cin>>n;
int A[n];
for(int i=0;i {
cin>>A[i];
}
selectionSort(A,n);
cout<<"Sorted Array is\n";
for(int i=0;i < n;i++)
{
cout < < A[i]<<" ";
}
}
